## 메소드 기반 문자열 조합 (String formatting method calls)
- '__{}_{}__'.format(value, value) 스타일 문자열 조합

    ```python
    # ex1)
    fs = '{0}...{1}...{2}'
    ms = fs.format('Robot', 125, 'Box')
    ms
    'Robot...125...Box'
    # {0}은 'Robot'으로 대체됨
    # {1}은 125로 대체됨
    # {2}는 'Box'로 대체됨

    # ex2)
    '{0}...{1}...{2}'.format('Robot', 125, 'Box')
    'Robot...125...Box'

    # ex3)
    '{2}...{1}...{0}'.format('Robot', 125, 'Box')
    'Box...125...Robot'

    # ex4) 숫자 생략 가능
    '{}...{}...{}'.format('Robot', 125, 'Box')
    'Robot...125...Box'

    # ex5) 이름으로 대체할 대상 지정하는 방식
    '{toy}...{num}...{item}'.format(toy = 'Robot', num = 125, item = 'Box')
    'Robot...125...Box'

    # ex6)
    my = ['Robot', 125, 'Box']
    '{0}...{1}...{2}'.format(*my)  # 인자 전달 과정에서 리스트 대상으로 언패킹
    'Robot...125...Box'

    # ex7)
    my = ['Box', (24, 31)]
    '{0[0]}..{0[1]}..{1[0]}..{1[1]}'.format(*my)  # {0}은 'Box'이고 {1}은 (24, 31)이다.
    'B..o..24..31'
    # {0}은 'Box'이므로 {0[0]}은 'B', {0[1]}은 'o'
    # {1}은 (24, 31)이므로 {1[0]}은 24, {1[1]}은 31

    # ex8) 딕셔너리 기반의 인덱싱 연산
    d = {'toy': 'Robot', 'price': 3500}
    'toy = {0[toy]}, price = {0[price]}'.format(d)
    'toy = Robot, price = 3500'
    ```

- 보다 세밀한 문자열 조합 지정
    - %[flags][width][.precision]f

    ```python
    '{0}'.format(3.14)
    '3.14'
    # 위와 같이 {0}으로 3.14 출력하는 경우, 파이썬이 '실수(real number)'라고 판단 및 출력

    '{0:f}'.format(3.14)
    '3.140000'
    # 위와 같이 {0:f}으로 3.14 출력하는 경우, 프로그래머가 '실수'라고 직접 알려주는 것임
    # {0:f}와 같이 f를 붙여서 실수를 출력하면 정밀도가 소수 이하 6자리로 설정됨

    '{0:d}'.format(3)  # d는 값의 종류가 정수임을 명시하는 것임
    '3'

    '{0:f}'.format(3.1)  # f는 값의 종류가 실수임을 명시하는 것임
    '3.100000'

    '%f' % 3.14
    '3.140000'

    '{0:f}'.format(3.14)
    '3.140000'

    '%.4f' % 3.14  # % 뒤에 .4를 넣어 줌
    '3.1400'

    '{0:.4f}'.format(3.14)  # : 뒤에 .4를 넣어줌
    '3.1400'

    '%9.4f' % 3.14  # %와 .4 사이에 9를 넣어줌
    '   3.1400'

    '{0:9.4f}'.format(3.14)  # :와 .4 앞에 9를 넣어 줌
    '   3.1400'

    '{0:<10.4f}'.format(3.14)  # <은 왼쪽으로 붙이라는 의미
    '3.1400    '

    '{0:>10.4f}'.format(3.14)  # >은 오른쪽으로 붙이라는 의미
    '    3.1400'

    '{0:^10.4f}'.format(3.14)  # ^은 중앙에 위치시키라는 의미
    '  3.1400  '

    '%+d, %+d' % (5, -5)  # 부호 출력의 의미로 % 뒤에 + 넣어 줌
    '+5, -5'

    '{0:+d}m {1:+d}'.format(5, -5)  # 부호 출력의 의미로 : 뒤에 + 넣어 줌
    '+5m -5'
    # 위의 문장에서 d는 생략 가능, 즉 '{0:+}, {1:+}'.format(5, -5)와 같음
    # 그리고 순서대로 등장하니 0과 1도 생략 가능, 즉 '{:+}, {:+}'.format(5, -5)와 같음

    '{:+}, {:+}'.format(5, -5)
    '+5, -5'

    '{0:*^10.4f}'.format(3.14)  # :와 ^ 사이에 등장한 *로 빈 공간 채움
    '**3.1400**'

    '{0:+<10}'.format(7)  # :와 < 사이에 등장한 +로 빈 공간 채움
    '7+++++++++'

    '{0:^^10}'.format('hi')  # :와 ^ 사이에 등장한 ^으로 빈 공간 채움
    '^^^^hi^^^^'
    # 위의 세 문장 모두 0 생략 가능, 즉 마지막 문장은 '{:^^10}.format('hi')와 같음
    ```
